# Docker com o ROS

# Referencia: https://medium.com/@sepideh.92sh/how-docker-revolutionizes-application-development-a-comprehensive-guide-for-beginners-fc2d3e53eb31


# Build the container and name it ros-noetic-container
docker build ~/Docker -t ros-noetic-container

# Container run
docker run -it --rm --user=$(id -u $USER):$(id -g $USER) --env="DISPLAY" --volume="/etc/group:/etc/group:ro" --volume="/etc/passwd:/etc/passwd:ro" --volume="/etc/shadow:/etc/shadow:ro" --volume="/etc/sudoers.d:/etc/sudoers.d:ro" --net host -v /home:/home -v ~/Volumes:/home/usr/ ros-noetic-updated:


# Start the terminator application for better terminal window management
terminator -u

# Verify that you're in a Docker container running Ubuntu 20.04
lsb_release -a

# Verify that ROS is correctly installed in the container
roscore

# Save docker container
# Necessario quando forem instalados novos pacotes dentro do container

docker ps

docker commit ros-noetic-updated:v3 ros-noetic-updated:v4
